while文
  • 例文:
    #include
using namespace std;

int main(){//やっぱり0から9までたしつづけてくれるプログラム
	int n=0;
	int i=0;
	while(i<10){
		n=n+i;
		i++;//iを1増やす
	}
	cout << n << endl;

	return 0;
}
  • たとえばwhile(1){}とかにすると∞ループができる
  • breakっていう子がいる。そいつをみつけるとループから抜ける。以下例文
    #include
using namespace std;

int main(){//やっぱり0から9までたしつづけてくれるプログラム
	int n=0;
	int i=0;
	while(1){//これだけだと∞ループ。しかし・・・
		n=n+i;
		i++;//iを1増やす
		if(i==10){//iが１０になったら
			break;//ループをぬけろと
		}
	cout << n << endl;

	}

	return 0;
}
